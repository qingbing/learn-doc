Node.js - 文件系统模块 fs-目录操作

1. 创建目录

  1. 异步创建目录

    // 声明文件系统
let fs = require('fs');

// 异步创建目录
fs.mkdir('./testdir', function (err) {
    if (err) {
        throw err;
    }
});

  2. 同步创建目录

    // 声明文件系统
let fs = require('fs');

// 同步创建目录
fs.mkdirSync('./testdi1r');
2. 读取文件目录下的目录及文件列表

  1. 异步读取目录下文件目录及文件列表

    // 声明文件系统
let fs = require('fs');

// 异步读取目录下文件目录及文件列表
fs.readdir('./testdir', function (err, list) {
    if (err) {
        throw  err;
    } else {
        console.log(list);
    }
});

  2. 同步读取目录下文件目录及文件列表

    // 声明文件系统
let fs = require('fs');

// 同步读取目录下文件目录及文件列表
var list = fs.readdirSync('./testdir');
console.log(list);
3. 删除空目录

  1. 异步删除空目录

    // 声明文件系统
let fs = require('fs');

// 异步删除空目录
fs.rmdir('./11', function (err, data) {
    if (err) {
        throw  err;
    } else {
        console.log(data);
    }
});

  2. 同步删除空目录

    // 声明文件系统
let fs = require('fs');

// 同步删除空目录
var status = fs.rmdirSync('./22');
console.log(status);
