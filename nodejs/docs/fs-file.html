Node.js - 文件系统模块 fs-文件操作

1. 什么是文件系统模块

由于nodejs是服务端运行的程序，需要提供服务器文件操作的模块，就是文件系统模块fs。

2. 读文件

提供两种读取文件的方式

  1. 直接读取 ： 将文件中的所有内容全部读取到内存后才触发回调函数

    1. 异步读取

      // 声明文件系统
let fs = require('fs');

// 异步读取文件内容
fs.readFile("./main.js", function (err, data) {
    if (err) {
        console.log("读取文件错误" + err);
    } else {
        console.log(data.toString());
    }
});

    2. 同步读取

      // 声明文件系统
let fs = require('fs');

// 同步读取文件内容
let data = fs.readFileSync("./main.js");
if(data){
    console.log(data.toString());
}

3. 写入文件

  1. 异步写入

    // 声明文件系统
let fs = require('fs');

// 异步写入文件内容
fs.writeFile("./index.html", "Hello node js!", function (err) {
    if (err) {
        console.log(err);
    } else {
        console.log("文件写入成功");
    }
});

  2. 同步写入

    // 声明文件系统
let fs = require('fs');

// 同步写入文件内容，返回undefined
fs.writeFileSync("./index.html", "Hello world!");
4. 读取文件信息

  1. 异步读取文件信息

    // 声明文件系统
let fs = require('fs');

// 异步写入文件内容
fs.stat('./main.js', function (err, stat) {
    if (err) {
        throw err;
    } else {
        console.log(stat);
    }
});

  2. 同步读取文件信息

    // 声明文件系统
let fs = require('fs');

// 异步读取文件信息
var stat = fs.statSync('./main.js');
console.log(stat);
5. 删除文件

  1. 异步删除文件

    // 声明文件系统
let fs = require('fs');

// 异步删除文件
fs.unlink('./11.js', function (err) {
    if (err) {
        throw  err;
    } else {
        console.log("文件删除成功");
    }
});

  2. 同步删除文件

    // 声明文件系统
let fs = require('fs');

// 异步读取文件信息
var status = fs.unlinkSync('./22.js');
console.log(status);
