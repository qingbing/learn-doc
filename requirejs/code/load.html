require —— 三种加载方式

一、配置文件方式（推荐）

<script data-main="./js/config" src="./js/require.js"></script>

二、配置变量前置（变量必须为require，放在引入require.js之前）

<script>

    var require = {

        deps: ["js/helper/util"],

        callback: function(util) {

            console.log(util);

        }

    };

</script>

<script src="./js/require.js"></script>

三、引入require.js后，直接require([rely],callback)

<script src="./js/require.js"></script>

<script>

    require(["js/helper/util", ...], function (util, ...) {

        console.log(util);

    });

</script>
