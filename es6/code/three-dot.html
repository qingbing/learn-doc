ES6 —— 三点运算符的使用

1. 认识三点运算符

用途

  1. rest（可变，剩下所有）参数
    用来取代 arguments， 但比arguments 灵活，只能是最后部分形式参数(arguments是伪数组，不能使用array的常规方法)
  2. 扩展运算符
    相当于把数组展开表示
// ... 表示所有参数
let fun5 = function (...val) {
    val.forEach((item, value) => {
        console.log(item, value);
    })
};

// ... 表示除了前面显示定义变量外剩下的所有参数
let fun6 = function (name, age, ...val) {
    val.forEach((item, value) => {
        console.log(item, value);
    })
};

fun5(2, 3);
fun6('qingbing', '19', 3, 4, 5, 6);

let arr = [1, 6];
let arr1 = [2, 3, 4, 5];
// ... 相当于遍历后插入
arr = [arr[0], ...arr1, arr[1]];
console.log(arr);
// ... 相当于遍历打印
console.log(...arr);
2. 将一个数组，变为参数序列
// 将一个数组，变为参数序列
let add = (x, y) => x + y;
let numbers = [3, 45];
console.log(add(...numbers));
3. 使用扩展运算符展开数组代替apply方法，将数组转为函数的参数
//ES5取数组最大值
console.log(Math.max.apply(this, [654, 233, 727]));
//ES6扩展运算符
console.log(Math.max(...[654, 233, 727]));
4. 使用push将一个数组添加到另一个数组的尾部
// ES5  写法
var arr1 = [1, 2, 3];
var arr2 = [4, 5, 6];
Array.prototype.push.apply(arr1, arr2);
//push方法的参数不能是数组，通过apply方法使用push方法
// ES6  写法
let arr1 = [1, 2, 3];
let arr2 = [4, 5, 6];
arr1.push(...arr2);
5. 合并数组
var arr1 = ['a', 'b'];
var arr2 = ['c'];
var arr3 = ['d', 'e'];
// ES5 的合并数组
var new_arr = arr1.concat(arr2, arr3);
// [ 'a', 'b', 'c', 'd', 'e' ]
// ES6 的合并数组
let new_arr = [...arr1, ...arr2, ...arr3];
6. 将字符串转换为数组
//ES5
var str = "hello";
var arr10 = str.split('');
console.log(arr10);

//ES6
let arr11 = [...'hello'];
console.log(arr11);
7. 转伪数组为真数组
var nodeList = document.querySelectorAll('p');
var array = [...nodeList];
8. map结构
// map结构
let map = new Map([
    [1, 'one'],
    [2, 'two'],
    [3, 'three'],
]);
let arr = [...map.keys()];
