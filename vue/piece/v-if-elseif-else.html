Vue v-if-elseif-else 指令示例

html-template 
<div v-if="score>90">优秀 {{score}}</div>
<div v-else-if="score > 80">良好 {{score}}</div>
<div v-else-if="score > 60">及格 {{score}}</div>
<div v-else>不及格 {{score}}</div>
new Vue 
let data = {
    "score": "78"
};
let myVue = new Vue({
    el: '#app',
    data
});
